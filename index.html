grade == 'A';
if (percentage >= 90 && percentage < 100 ) {
grade = 'A';
}

} else if (percentage >= 80 && percentage < 90 ) {

grade = 'C';
} else if (percentage >= 60 && percentage < 70) {
grade = 'D';
} else if (percentage >= 50 && percentage < 60) {
grade = 'E';
} else {
grade = 'F';
}

}
